Node & Edge Schema

Review each kind's properties — let me know what to add, remove, or rename

Nodes — type TsNode = FolderNode | FileNode | FunctionNode | ClassNode | InterfaceNode | ImportNode

FolderNode 
kind: 'FOLDER'
id: string        // absolute path
path: string
name: string
depth: number     // 0 = repo root
fileCount: number
childFolderCount: number
FileNode 
kind: 'FILE'
id: string        // absolute path
path: string
name: string
extension: string  // .ts | .tsx | .d.ts
isEntrypoint: boolean
isBarrel: boolean  // index.ts re-exporting
importCount: number
exportCount: number
lineCount: number
FunctionNode 
kind: 'FUNCTION'
id: string        // path#name
name: string
filePath: string
isExported: boolean
isAsync: boolean
isArrow: boolean
paramCount: number
returnType: string | null
ClassNode 
kind: 'CLASS'
id: string        // path#ClassName
name: string
filePath: string
isExported: boolean
isAbstract: boolean
methodCount: number
propertyCount: number
extends: string | null
implements: string[]
InterfaceNode 
kind: 'INTERFACE'
id: string        // path#InterfaceName
name: string
filePath: string
isExported: boolean
propertyCount: number
methodCount: number
extends: string[]
ImportNode 
kind: 'IMPORT'
id: string        // moduleSpecifier
moduleSpecifier: string
name: string
isExternal: boolean
isTypeOnly: boolean
namedImports: string[]

Edges — type TsEdge = ImportEdge | ExportEdge

ImportEdge 
type: 'import'
source: string    // FileNode id
target: string    // ImportNode id
isTypeOnly: boolean
namedImports: string[]
defaultImport: string | null
namespaceImport: string | null
ExportEdge 
type: 'export'
subtype: 'declaration' | 'reexport'
source: string    // FileNode id
target: string    // Function|Class|Interface|FileNode id
isTypeOnly: boolean
exportedName: string | null  // alias
