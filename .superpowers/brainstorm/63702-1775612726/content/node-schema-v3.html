Node & Edge Schema

All nodes share a base — each kind extends it with its own properties

Shared Base — TsNodeBase

id: string          // unique node id
kind: NodeKind
parent: string | null  // parent node id
children: string[]    // child node ids
siblings: string[]    // sibling node ids (excl. self)

type TsNode = FolderNode | FileNode | FunctionNode | ClassNode | InterfaceNode | ImportNode
Each kind is TsNodeBase & { kind: '...', ...kind-specific props }

Node Kinds

FolderNode 
// ...TsNodeBase
kind: 'FOLDER'
path: string
name: string
depth: number
fileCount: number
childFolderCount: number

// parent  = parent FOLDER | null
// children = sub-FOLDERs + FILEs
// siblings = items in same FOLDER
FileNode 
// ...TsNodeBase
kind: 'FILE'
path: string
name: string
extension: string
isEntrypoint: boolean
isBarrel: boolean
importCount: number
exportCount: number
lineCount: number

// parent  = containing FOLDER
// children = FUNCTIONs, CLASSes, INTERFACEs
// siblings = files in same FOLDER
FunctionNode 
// ...TsNodeBase
kind: 'FUNCTION'
name: string
filePath: string
isExported: boolean
isAsync: boolean
isArrow: boolean
paramCount: number
returnType: string | null

// parent  = containing FILE or CLASS
// children = nested functions (if any)
// siblings = other top-level decls in FILE
ClassNode 
// ...TsNodeBase
kind: 'CLASS'
name: string
filePath: string
isExported: boolean
isAbstract: boolean
methodCount: number
propertyCount: number
extends: string | null
implements: string[]

// parent  = containing FILE
// children = methods + properties
// siblings = other top-level decls in FILE
InterfaceNode 
// ...TsNodeBase
kind: 'INTERFACE'
name: string
filePath: string
isExported: boolean
propertyCount: number
methodCount: number
extends: string[]

// parent  = containing FILE
// children = property/method signatures
// siblings = other top-level decls in FILE
ImportNode 
// ...TsNodeBase
kind: 'IMPORT'
moduleSpecifier: string
name: string
isExternal: boolean
isTypeOnly: boolean
namedImports: string[]

// parent  = FOLDER if in-project, null if external
// children = named exports (if tracked)
// siblings = other imports in same scope

Edges — type TsEdge = ImportEdge | ExportEdge

ImportEdge 
type: 'import'
source: string    // FileNode id
target: string    // ImportNode id
isTypeOnly: boolean
namedImports: string[]
defaultImport: string | null
namespaceImport: string | null
ExportEdge 
type: 'export'
subtype: 'declaration' | 'reexport'
source: string    // FileNode id
target: string    // Function|Class|Interface|FileNode id
isTypeOnly: boolean
exportedName: string | null  // alias
